# Genius Engineer found
character_event = {
	id = RIP.4501
	desc = EVTDESC_ATLA_RIP_4501
	picture = GFX_evt_fire_factory
	border = GFX_event_normal_frame_religion

	min_age = 16
	capable_only = yes
	only_playable = yes
	prisoner = no
	war = no
	has_dlc = "Reapers"

	is_triggered_only = yes

	trigger = {
		higher_tier_than = BARON
		any_demesne_province = {
			AND = {
				OR = {
					hospital_level = 1
					hospital_level = 2
					hospital_level = 3
					hospital_level = 4
					hospital_level = 5
					hospital_level = 6
				}
				religion = ROOT
			}
		}
		OR = {
			NOT = {
				any_courtier_or_vassal = {
					has_minor_title = title_commander
				}
			}
			any_courtier_or_vassal = {
				has_minor_title = title_commander
				NOT = { martial = 9 }
			}
		}
	}

	immediate = {
		any_courtier_or_vassal = {
			limit = {
				has_minor_title = title_commander
				NOT = { martial = 9 }
			}
			save_event_target_as = previous_physician
		}
		random_demesne_province = {
			limit = {
				AND = {
					OR = {
						hospital_level = 1
						hospital_level = 2
						hospital_level = 3
						hospital_level = 4
						hospital_level = 5
						hospital_level = 6
					}
					religion = ROOT
				}
			}
			save_event_target_as = pilgrim_physician_hospital
		}
		create_character = {
			religion = ROOT
			culture = ROOT
			dynasty = none
			age = 29
			random_traits = yes
			female = no
			attributes = {
				martial = 7
			}
			health = 6
			trait = tough_soldier
			trait = siege_leader
			trait = shrewd
			trait = patient
		}
		new_character = {
			remove_trait = scholar
			remove_trait = proud
			remove_trait = socializer
			remove_trait = gamer
			remove_trait = hedonist
			remove_trait = duelist
			remove_trait = hunter
			remove_trait = strategist
			remove_trait = administrator
			remove_trait = architect
			remove_trait = gardener
			remove_trait = schemer
			remove_trait = impaler
			remove_trait = seducer
			remove_trait = seductress
			remove_trait = mystic
			save_event_target_as = pilgrim_physician
		}
	}

	option = { # Appoint the Engineer as a commander
		name = EVTOPTA_ATLA_RIP_4501
		event_target:previous_physician = {
			remove_title = title_commander
		}
		event_target:pilgrim_physician = {
			give_minor_title = title_commander
		}
	}
	option = { # Ignore him
		name = EVTOPTB_ATLA_RIP_4501
		hidden_tooltip = {
			event_target:pilgrim_physician = {
				death = {
					death_reason = death_missing
				}
			}
		}
	}
}
